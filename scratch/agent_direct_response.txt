#### `snake.py`
```python
import pygame
import random

class Snake:
    def __init__(self):
        self.body = [(100, 100), (90, 100), (80, 100)]
        self.direction = (10, 0)
        self.grow = False

    def update(self):
        if self.grow:
            self.body.insert(0, (self.body[0][0] + self.direction[0], self.body[0][1] + self.direction[1]))
            self.grow = False
        else:
            self.body.insert(0, (self.body[0][0] + self.direction[0], self.body[0][1] + self.direction[1]))
            self.body.pop()

    def change_direction(self, direction):
        if (direction[0] * -1, direction[1] * -1) != self.direction:
            self.direction = direction

    def grow_snake(self):
        self.grow = True

    def get_head_position(self):
        return self.body[0]

    def get_body(self):
        return self.body[1:]

    def reset(self):
        self.__init__()

    def draw(self, surface):
        for segment in self.body:
            pygame.draw.rect(surface, (0, 255, 0), pygame.Rect(segment[0], segment[1], 10, 10))
```

#### `food.py`
```python
import pygame
import random

class Food:
    def __init__(self):
        self.position = (0, 0)
        self.randomize_position()

    def randomize_position(self):
        self.position = (random.randint(0, 39) * 10, random.randint(0, 29) * 10)

    def get_position(self):
        return self.position

    def draw(self, surface):
        pygame.draw.rect(surface, (255, 0, 0), pygame.Rect(self.position[0], self.position[1], 10, 10))
```

#### `game.py`
```python
import pygame
from snake import Snake
from food import Food

class Game:
    def __init__(self):
        self.width = 400
        self.height = 300
        self.surface = pygame.display.set_mode((self.width, self.height))
        pygame.display.set_caption('Snake Game')
        self.clock = pygame.time.Clock()
        self.snake = Snake()
        self.food = Food()
        self.score = 0
        self.speed = 10

    def run(self):
        running = True
        while running:
            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    running = False
                elif event.type == pygame.KEYDOWN:
                    if event.key == pygame.K_UP:
                        self.snake.change_direction((0, -10))
                    elif event.key == pygame.K_DOWN:
                        self.snake.change_direction((0, 10))
                    elif event.key == pygame.K_LEFT:
                        self.snake.change_direction((-10, 0))
                    elif event.key == pygame.K_RIGHT:
                        self.snake.change_direction((10, 0))

            self.snake.update()
            if self.snake.get_head_position() == self.food.get_position():
                self.snake.grow_snake()
                self.food.randomize_position()
                self.score += 1
                self.speed += 1

            if (self.snake.get_head_position()[0] < 0 or
                self.snake.get_head_position()[0] >= self.width or
                self.snake.get_head_position()[1] < 0 or
                self.snake.get_head_position()[1] >= self.height or
                self.snake.get_head_position() in self.snake.get_body()):
                self.snake.reset()
                self.score = 0
                self.speed = 10

            self.surface.fill((0, 0, 0))
            self.snake.draw(self.surface)
            self.food.draw(self.surface)
            pygame.display.flip()
            self.clock.tick(self.speed)
```
